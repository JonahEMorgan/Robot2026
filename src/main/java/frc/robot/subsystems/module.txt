turn math
normal force (lb) <- robot weight / number of wheels
contact patch length (in) <- normal force / wheel stiffness
characteristic speed (ft/s) <- contact patch length / 12 in/ft / rubber relaxation time
current speed (ft/s) <- .5 * drive velocity * wheel diameter / 12 in/ft
torque from friction (in*lb) <- sqrt((contact patch length)^2 + (wheel width)^2) * coefficient of friction * normal force / 3 / exp(current speed / characteristic speed)
moment of inertia (lb*in^2) <- wheel weight * (.75 * (wheel diameter)^2 + (wheel width)^2) / 12
acceleration (rad/s^2) <- turn ka * (turn voltage - turn velocity / turn kv) / moment of inertia / turn resistance / turn gear ratio
velocity (rad/s) <- turn kv * (turn voltage - torque from friction * turn resistance / turn ka) / turn gear ratio

drive math
normal force (lb) <- robot weight / number of wheels
torque from rolling resistance (in*lb) <- .5 * wheel diameter * normal force * coefficient of rolling resistance
torque from robot (in*lb) <- .5 * wheel diameter * normal force
total torque (in*lb) <- torque from rolling resistance + torque from robot
moment of inertia (lb*in^2) <- wheel weight * (wheel diameter)^2 / 8
acceleration (rad/s^2) <- drive ka * (drive voltage - drive velocity / drive kv) / moment of inertia / drive resistance / drive gear ratio
velocity (rad/s) <- drive kv * (drive voltage - total torque * drive resistance / drive ka / drive gear ratio) / drive gear ratio
